function myAdd() {
  if (arguments.length == 0) {
    return;
  } 
  else if (arguments.length == 1) {
    let c = arguments[0];
    return function (arg2) {
      return c + arg2;
    } 
  }
  else {
    return arguments[0] + arguments[1];
  }
}
  
console.log(myAdd(20,20)); //40
console.log(myAdd(10)(10));  //20
console.log(myAdd());   //undefined
