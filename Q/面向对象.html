function A(){
	this.sayHello3 = function(){
		console.log("sayHello3");
	}
}

A.sayHello1=function(){
	console.log("sayHello1");
}

A.prototype.sayHello2=function(){
	console.log("sayHello2");
}

var a=new A();

A.sayHello1();  //sayHello1

a.sayHello2();  //sayHello2

//继承
function B(){

}

B.prototype = new A();
var b = new B();
b.sayHello2();  //sayHello2
b.sayHello3();  //sayHello3
